#
#   macOS / Linux:
#     .venv/bin/python3 -m pip install -U pip
#     .venv/bin/python3 -m pip install -r requirements.txt
#
# --- Run the GUI ---
#
#   Windows:    .venv\Scripts\python.exe XDM1041_GUI.py
